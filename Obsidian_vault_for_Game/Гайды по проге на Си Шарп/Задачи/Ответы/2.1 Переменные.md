//прямоугольник
	int a = 10;
    int b = 20;
    int Surface = a * b;
    Console.WriteLine("Площать прямоугольника: " + Surface + " кв.см");
            
// Круг
	int radius = 13;
	float pi = 3.14f;
	float Surface = pi * radius * radius; // Так как мы используем дробное число пи в рассчетах, то  и ответ должен быть дробного типа

	Console.WriteLine("Площать круга: " + Surface + " кв.см");
	
// Треугольник
	 int a = 11;
	 int h = 5;
	 float Surface = (a * h) / 2f; 
	 // При делении площать может получится дробным числом, поэтому и тип перменной и делитель должны быть дробными

	 Console.WriteLine("Площать треугольника: " + Surface + " кв.см");
	 
// Квадрат
	 int a = 11;
	 int h = 5;
	 float Surface = (a * h) / 2f; 
	 // При делении площать может получится дробным числом, поэтому и тип перменной и делитель должны быть дробными

	 Console.WriteLine("Площать треугольника: " + Surface + " кв.см");
