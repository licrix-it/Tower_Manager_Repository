```C#
//Для того чтобы выполнять код при определенных условиях мы используем кострукцию if/else.
// if (Условие) {Код который будем выполнять, если условие верно} else {Код, который выполнится если условие НЕ верно}
// Условием является любое выражение, которое может вернуть значение типа boolean. Для этого есть базовые операторы сравнения
// == != > < >= <= (Равно, не равно, больше, меньше, больше или равно, меньше или равно)

int age = 18;
if (age >= 18) // Читаем буквально как написано (если age > 18), тогда {}
{
	Console.WriteLine("Можешь водить машину");
	// Данный блок кода выполнится, если age Имеет любое значение больше 18
}
else
{
	Console.WriteLine("Никаких тебе прав");
	// Данный блок кода выполнится во всех остальных случаях, т.е 18 и меньше 18. Отдельный if можно не писать для этого
}
-------------------------------------------------------------------------------------------------------------------------
// Есил в блоке кода {} Всего одна команда, то можно скобки не писать (Но все равно рекомендуется их писать везде, просто потому что так проще)

if (age >= 18) 
	Console.WriteLine("Можешь водить машину");
else
	Console.WriteLine("Никаких тебе прав");
--------------------------------------------------------------------------------------------------------------------------
// Если нужна более сложная проверка, можно стакать условия else if
// Допустим у нас всего 3 случая, больше 18, ровно 18 и меньше 18, четвертого тут в принципе быть не может. Значит можно использовать 2 условия, а третий просто оставить в else.

int age = 18;

if (age > 18)
{
	Console.WriteLine("Тебе точно БОЛЬШЕ 18"); 
}
else if(age == 18)
{
	Console.WriteLine("Тебе РОВНО 18");
}
else 
{
	Console.WriteLine("Тебе точно Меньше 18");
}
---------------------------------------------------------------------------------------------------------------------------------
// Условий можнор стакать сколько угодно, просто пиши else if друг за дружкой. Если надо 5 условий, то будет 4 if, а 5 условие в остаток через else


string race = "human";

if (race == "human")
{
	Console.WriteLine("Вы, человек, уже неплохо для начала"); 
}
else if (race = "monkey")
{
	Console.WriteLine("Иди еще поэволюционируй немного, примат"); 
}
else if (race = "elf")
{
	Console.WriteLine("Я что исекайнулся?"); 
}
else if(race = "Dwarf")
{
	Console.WriteLine("Где пиво?!");
}
else 
{
	Console.WriteLine("Я вообщей хуй знает какой я рассы");
}
------------------------------------------------------------------------------------------------------------------------------------
// Условия можно объединять с помощью логических операторов && || ! (И, ИЛИ, НЕ)

int age = 18;
string race = "monkey";

if (age >= 18 && race = "monkey")
{
	Console.WriteLine("Так ты уже совершеннолетняя обезьяна");
	/*
		Сравнение множества условий происходит попорядку
		1. age >=18 (результат true)
		2. race = "monkey" (результат true)
		3. ture && true (результат true) Логический оператор сравнивает не изначальные переменные, а результат их условий
		   4. Если нам меньше 18, то первое условие будет false. тогда оператор *И* будет работать так: false && true (результат false)
	*/
}

```
